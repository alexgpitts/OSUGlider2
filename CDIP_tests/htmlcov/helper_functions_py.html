47def Plotter(fig, axs, xy) -> NULL:

77def Rolling_mean(x: np.array, w: np.array) -> np.array: 

78 """Smoothes the raw acceleration data with a rolling mean.  

79 Accepts data to be smoothed and a window width for the moving average.  

80 """ 

81 df = pd.DataFrame({'a': x.tolist()}) 

82 return df['a'].rolling(window=w, center=True).mean().fillna(0).to_numpy()

86def Bias(width: int, window: str = "hann") -> np.array: 

87 """returns a either a boxcar, or hann window""" 

88 return np.ones(width) if window == "boxcar" else ( 

89 0.5*(1 - np.cos(2*np.pi*np.arange(width) / (width - 0))) 

90 )

93def wfft(data: np.array, width: int, window: str = "hann") -> list[np.array]: 

94 """Splits the acceleration data into widows,  

95 preforms FFTs on them returning a list of all the windows 

96 """ 

97 bias = Bias(width, window) 

98 

99 ffts = [] 

100 for i in range(0, data.size-width+1, width//2): 

101 w = data[i:i+width] 

102 ffts.append(np.fft.rfft(w*bias)) 

103 

104 return ffts

107def wcalcPSD( 

108 A_FFT_windows: list[np.array], 

109 B_FFT_windows: list[np.array], 

110 fs: float, 

111 window: str) -> np.array: 

112 """calculates the PSD of the FFT output preformed with the windowing method. 

113 After calculateing the PSD of each window, the resulting lists are averaged together""" 

114 

115 width = A_FFT_windows[0].size 

116 spectrums = np.complex128(np.zeros(width)) 

117 for i in range(len(A_FFT_windows)): 

118 A = A_FFT_windows[i] 

119 B = B_FFT_windows[i] 

120 

121 spectrum = calcPSD(A, B, fs, window=window) 

122 spectrums += spectrum 

123 return spectrums / len(A_FFT_windows)

126def calcPSD(xFFT: np.array, yFFT: np.array, fs: float, window: str) -> np.array: 

127 "calculates the PSD on an output of a FFT" 

128 nfft = xFFT.size 

129 qOdd = nfft % 2 

130 n = (nfft - qOdd) * 2 # Number of data points input to FFT 

131 w = Bias(n, window) # Get the window used 

132 wSum = (w * w).sum() 

133 psd = (xFFT.conjugate() * yFFT) / (fs * wSum) 

134 if not qOdd: # Even number of FFT bins 

135 psd[1:] *= 2 # Real FFT -> double for non-zero freq 

136 else: # last point unpaired in Nyquist freq 

137 psd[1:-1] *= 2 # Real FFT -> double for non-zero freq 

138 return psd

141def calcAcceleration(x: np.array, fs: float) -> np.array: 

142 """converts displacement data to acceleration. 

143 We need acceleration data because that is 

144 what we will record from the STM""" 

145 dx2 = np.zeros(x.shape) 

146 dx2[2:] = np.diff(np.diff(x)) 

147 dx2[0:2] = dx2[2] 

148 return dx2 * fs * fs

151def Data(filename) -> dict: 

152 """Master data reading function. Reads the .nc file from CDIP. 

153 The data is stored in dictionary (data), which contains many dictionaries  

154 to hold information. Examples include: acceleration data, frequency bounds,  

155 expected values calculated by CDIP, etc.""" 

156 meta_xr = xr.open_dataset(filename, group="Meta") # For water depth 

157 wave_xr = xr.open_dataset(filename, group="Wave") 

158 xyz_xr = xr.open_dataset(filename, group="XYZ") 

159 

160 depth = float(meta_xr.WaterDepth) 

161 declination = float(meta_xr.Declination) 

162 frequency = float(xyz_xr.SampleRate) 

163 

164 data = {} 

165 if META: 

166 data["frequency"] = frequency 

167 data["latitude"] = float(meta_xr.DeployLatitude) 

168 data["longitude"] = float(meta_xr.DeployLongitude) 

169 data["depth"] = depth 

170 data["declination"] = declination 

171 

172 if TXYZ: 

173 data["time"] = xyz_xr.t.to_numpy() 

174 data["dis"] = { 

175 "t": data["time"], 

176 "x": xyz_xr.x.to_numpy(), 

177 "y": xyz_xr.y.to_numpy(), 

178 "z": xyz_xr.z.to_numpy() 

179 } 

180 

181 data["acc"] = { 

182 "t": data["time"], 

183 "x": calcAcceleration(xyz_xr.x.to_numpy(), frequency), 

184 "y": calcAcceleration(xyz_xr.y.to_numpy(), frequency), 

185 "z": calcAcceleration(xyz_xr.z.to_numpy(), frequency), 

186 } 

187 

188 if WAVE: 

189 data["wave"] = { 

190 "sig-height": wave_xr.Hs.to_numpy(), 

191 "avg-period": wave_xr.Ta.to_numpy(), 

192 "peak-period": wave_xr.Tp.to_numpy(), 

193 "mean-zero-upcross-period": wave_xr.Tz.to_numpy(), 

194 "peak-direction": wave_xr.Dp.to_numpy(), 

195 "peak-PSD": wave_xr.PeakPSD.to_numpy(), 

196 "a1": wave_xr.A1.to_numpy(), 

197 "b1": wave_xr.B1.to_numpy(), 

198 "a2": wave_xr.A2.to_numpy(), 

199 "b2": wave_xr.B2.to_numpy(), 

200 } 

201 

202 # data["wave"]["time-bounds"] = { 

203 data["time-bounds"] = { 

204 "lower": wave_xr.TimeBounds[:, 0].to_numpy(), 

205 "upper": wave_xr.TimeBounds[:, 1].to_numpy() 

206 } 

207 

208 if FREQ: 

209 data["freq"] = { 

210 "bandwidth": wave_xr.Bandwidth.to_numpy(), 

211 } 

212 

213 data["freq"]["bounds"] = { 

214 "lower": wave_xr.FreqBounds[:, 0].to_numpy(), 

215 "upper": wave_xr.FreqBounds[:, 1].to_numpy(), 

216 "joint": wave_xr.FreqBounds[:, :].to_numpy() 

217 } 

218 

219 return data
