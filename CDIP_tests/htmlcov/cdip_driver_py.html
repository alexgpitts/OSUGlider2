161 def welch(run: bool): 

162 if run == False: 

163 return 

164 

165 a0 = wPSD["zz"][1:] / np.square(np.square(2 * np.pi * wPSD["freq_space"][1:])) 

166 

167 m0 = (a0 * wPSD["freq_space"][1]).sum() 

168 m1 = (a0*wPSD["freq_space"][1:]*wPSD["freq_space"][1]).sum() 

169 

170 mm1 = (a0/wPSD["freq_space"][1:]*wPSD["freq_space"][1]).sum() 

171 te = mm1/m0 # mean energy period 

172 

173 m2 = (a0*np.square(wPSD["freq_space"][1:]) * wPSD["freq_space"][1]).sum() 

174 

175 tp = 1/wPSD["freq_space"][1:][a0.argmax()] 

176 

177 denom = np.sqrt(wPSD["zz"] * (wPSD["xx"] + wPSD["yy"])) 

178 a1 = wPSD["zx"].imag / denom 

179 b1 = -wPSD["zy"].imag / denom 

180 denom = wPSD["xx"] + wPSD["yy"] 

181 

182 dp = np.arctan2(b1[a0.argmax()], a1[a0.argmax()]) # radians 

183 

184 outputs[i]["welch"] = { 

185 "Hs": 4 * np.sqrt(m0), 

186 "Ta": m0/m1, #average period 

187 "Tp": tp, # peak wave period 

188 "wave_energy_ratio": te/tp, 

189 "Tz": np.sqrt(m0/m2), 

190 "Dp": np.arctan2(b1[a0.argmax()], a1[a0.argmax()]), 

191 "PeakPSD": a0.max(), 

192 "te": te, # mean energy period 

193 "dp_true": np.degrees(dp) % 360, 

194 "dp_mag": np.degrees(dp+data["declination"]) % 360, 

195 "a1": a1, 

196 "b1": b1, 

197 "a2": (wPSD["xx"] - wPSD["yy"]) / denom, 

198 "b2": -2 * wPSD["xy"].real / denom, 

199 }
